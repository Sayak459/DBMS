  EID NUMBER(5) PRIMARY KEY,
  ENAME VARCHAR2(30),
  CITY VARCHAR2(20),
  DESIGNATION VARCHAR2(20),
  SALARY NUMBER(10,2),
  PERKS NUMBER(10,2));

CREATE OR REPLACE TRIGGER update_salary_trigger
BEFORE INSERT ON EMP3
FOR EACH ROW
BEGIN
    IF :NEW.PERKS > 500 THEN
        :NEW.SALARY := :NEW.SALARY - 300;
    END IF;
END;

CREATE OR REPLACE TRIGGER update_salary_trigger
BEFORE UPDATE ON EMP3
FOR EACH ROW
BEGIN
    IF :NEW.SALARY<=500 THEN
        :NEW.SALARY := 10000;
    ELSE
        :NEW.SALARY := 15000;
    END IF;
END;
/

CREATE TABLE EMP4(
  EID NUMBER(5) PRIMARY KEY,
  ENAME VARCHAR2(30),
  SALARY NUMBER(10,2));

CREATE TABLE EMP_BACKUP(
  EID NUMBER(5) PRIMARY KEY,
  ENAME VARCHAR2(30),
  SALARY NUMBER(10,2));

CREATE TRIGGER emp_insert_trigger
AFTER INSERT ON EMP4
FOR EACH ROW
BEGIN
    INSERT INTO EMP_BACKUP(EID,ENAME,SALARY) VALUES (:new.EID, :new.ENAME, :new.SALARY);
END;
/
