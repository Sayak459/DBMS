create table ZIPCODES(
    ZIP NUMBER(10) PRIMARY KEY,
    CITY VARCHAR2(20)
);

create table EMPLOYEES(
    ENO NUMBER(10) PRIMARY KEY,
    ENAME VARCHAR2(20),
    ZIP NUMBER(10) FOREIGN KEY REFERENCES ZIPCODES(ZIP),
    HDATE DATE
);

create table PARTS(
    PNO NUMBER(10) PRIMARY KEY,
    PNAME VARCHAR2(20),
    QOH NUMBER(5),
    PRICE NUMBER(10,2),
    LEVEL NUMBER(5)
);

create table CUSTOMERS(
    CNO NUMBER(5) PRIMARY KEY,
    CNAME VARCHAR2(20),
    STREET VARCHAR2(40),
    ZIP NUMBER(10) FOREIGN KEY REFERENCES ZIPCODES(ZIP),
    PHONE NUMBER(14)
);

create table ORDERS(
    ONO NUMBER(10) PRIMARY KEY,
    CNO NUMBER(5) FOREIGN KEY REFERENCES CUSTOMERS(CNO),
    ENO NUMBER(10) FOREIGN KEY REFERENCES EMPLOYEES(ENO),
    PRECEIVED NUMBER(1),
    SHIPPED NUMBER(1)
);

create table ODETAILS(
    ONO NUMBER(10),
    PNO NUMBER(10),
    OTY NUMBER(5),
    PRIMARY KEY (ONO,PNO),
    FOREIGN KEY ONO REFERENCES ORDERS(ONO),
    FOREIGN KEY PNO REFERENCES PARTS(PNO)
);
